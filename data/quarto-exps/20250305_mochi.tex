\begin{frame}{ORCA: Brief Overview}

\begin{itemize}
\item
  ORCA: Observability with Realtime Control and Aggregation
\item
  Original motivation: programmable observability

  \begin{itemize}
  \tightlist
  \item
    Think of a helpful view over telemetry
  \item
    Express it as a SQL query
  \item
    Overlay will stream just the relevant data in real-time
  \end{itemize}
\end{itemize}

\pause

\begin{itemize}
\tightlist
\item
  Ingredients: on-fabric aggregation and real-time control

  \begin{itemize}
  \tightlist
  \item
    Interfaces to collect data as Arrow tables
  \item
    An overlay to move it around
  \item
    A distributed SQL engine to do telemetry aggregation
  \item
    Consistent control semantics for online probing, tuning etc.
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[fragile]{ORCA: Overlay Progress}
\phantomsection\label{orca-overlay-progress}
\begin{itemize}
\tightlist
\item
  Implemented Mercury-based Overlay

  \begin{itemize}
  \tightlist
  \item
    Addresses overlay entities using \texttt{ovid\_t}
  \item
    \texttt{ovid\_t}: 32-bit identifier, \textasciitilde{} rank for MPI
    ranks
  \item
    Special \texttt{ovid\_t} range for aggregators (unique for
    controller)
  \item
    Special values for broadcast, dedicated routes etc.
  \end{itemize}
\end{itemize}

\pause

\begin{itemize}
\tightlist
\item
  Bootstraps using a two-step process

  \begin{itemize}
  \tightlist
  \item
    Initial discovery using a TCP socket
  \item
    Bootstrap protocol using Mercury
  \item
    Can be extended for arbitrary topologies
  \end{itemize}
\end{itemize}

\pause

\begin{itemize}
\tightlist
\item
  Difference from \texttt{deltafs-shuffle}:

  \begin{itemize}
  \tightlist
  \item
    Purely an addressing and routing layer
  \item
    No batching, intra-MPI routing, delivery, flow control
  \item
    (Relegated to higher layers)
  \end{itemize}
\end{itemize}
\end{frame}
