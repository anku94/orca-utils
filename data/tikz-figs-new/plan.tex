% Define orcaflow language (SQL + YAML hybrid)
\lstdefinelanguage{orcaflow}{
    % SQL keywords
    keywords={SELECT, FROM, WHERE, AND},
    % YAML structure keywords
    morekeywords=[2]{nodes, sinks, sources, type, from, to, config, name, enabled, true, false},
    keywordstyle=\color{keyword}\bfseries,
    keywordstyle=[2]\color{keyword}\bfseries,
    sensitive=false,
    morecomment=[l]{\#},
    commentstyle=\color{comment}\itshape,
    morestring=[b]',
    morestring=[b]",
    stringstyle=\color{string},
}
