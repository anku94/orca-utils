\begin{columns}

  \column{0.5}

  \block{BSP Workloads Need Real-Time Observability}{
    \textbf{Problem:} BSP efficiency bottlenecks are primarily a diagnostic challenge
    \begin{itemize}
      \item BSP: Bulk-Synchronous Parallel, critical paradigm in HPC/ML
      \item Large jobs with O(10K) nodes, long-running (weeks), lockstep execution
            % workloads (HPC simulations, ML training) run in lockstep across O(10K) processors
      \item Uniquely fragile: single straggler blocks all workers
      \item Diverse root causes: fail-slow hardware, tuning artifacts, load imbalance
      \item Our experience: diagnosis takes months, mitigations take days
    \end{itemize}

    % \centering
    \raisebox{-0.5\height}{\includegraphics[height=4in, keepaspectratio]{lesson_volfunc_single_narrow.pdf}}%
    \hspace{1em}%
    \parbox[c]{8in}{\small Fig 1. A BSP pathology at 4096 ranks, with MPI\_Wait spikes blocking all ranks. Attribution requires joining fine-grained per-timestep telemetry.}

    \raggedright
    \textbf{Bottleneck:} post-hoc tracing workflows delay intervention
    \begin{itemize}
      \item Tracing: fine-grained telemetry emitted as per-rank logs
      \item Large datasets analyzed offline $\to$ delayed intervention
      \item Telemetry volume + structure $\to$ analytics bottleneck
    \end{itemize}

    \vspace{1em}
    \raisebox{-0.5\height}{\includegraphics[height=4in, keepaspectratio]{broken-loop.pdf}}%
    \hspace{1em}%
    \parbox[c]{7in}{\small Fig 2. Post-hoc trace analysis workflows produce large datasets, creating ETL (extract, transform, load) bottlenecks. Moreover, paradigm intrinsically precludes online intervention.}
  }

  \column{0.5}

  \block{Steerable Observability: Real-time Feedback and Control}{
    \textbf{Our Goal:} always-on BSP-aware observability --- lightweight metrics by default,
    with on-demand escalation and runtime intervention.
    % . Lightweight metrics in the common case,
    % with the ability to escalate collection, test hypotheses, apply interventions --- while the job runs.
    % . Lightweight metrics in the common case, depth on demand.
    % Diagnosis is iterative: need both observation and intervention
    % \textbf{Need:} always-on, steerable, BSP-aware observability
    % \begin{itemize}
    %   \item Lightweight metrics in the common case, depth on demand
    % \end{itemize}
    \vspace{1em}

    \textbf{Existing approaches and limitations:}
    \begin{itemize}
      \item \textbf{Monitoring} (Prometheus, LDMS): no workload context
      \item \textbf{Tracing} (TAU, Score-P, Kineto): post-hoc; volume delays feedback
      \item \textbf{Event Streaming} (Flink, Kafka): no BSP-aligned collection or control
    \end{itemize}

    \vspace{1em}
    \textbf{ORCA:} Observability with Realtime Control and Aggregation
    \begin{itemize}
      \item Coordinated collection of BSP-aligned telemetry as timestep dataframes
      \item In-situ reductions to materialize views directly from applications
      \item Control plane: asynchronous, atomic, timestep-consistent
    \end{itemize}

      Global synchronization provides natural points for observations and control.
    }
  }

\block{Architecture: Tree-based Overlay, Columnar In-situ Analytics, Timestep-Consistent Control}{
  \begin{minipage}[t]{0.49\linewidth}
    \vspace{0pt}
    \textbf{Transport: Incast-Optimized, On-Fabric, Receiver-Driven}
    \begin{itemize}
      \item Timestep dataframes: typed Arrow streams, flushed every timestep
      \item Aggregators retrieve asynchronously via RDMA GETs
    \end{itemize}

    \vspace{1em}
    \textbf{In-Situ Analytics: Columnar, Tiered Query Planning}
    \begin{itemize}
      \item OrcaFlow: SQL-based analytics en route to sinks
      \item DataFusion-based planner compiles to tier plans
      \item Sinks: DuckDB (real-time), timestep-partitioned Parquet (offline)
    \end{itemize}

    \vspace{1em}
    \textbf{Control Plane: Timestep-Aligned Two-Phase Commit (TS2PC)}
    \begin{itemize}
      \item Controller proposes commands for future timestep $T_{cmd} = T_{app} + \Delta$
      \item Commands committed only if no rank has passed $T_{cmd}$
      \item $\Delta$ learned automatically: initialized to 0, incremented on abort
    \end{itemize}
  \end{minipage}%

  \block{Tracing: Lower Overhead, Smaller Traces, Faster Queries}{
    % Evaluated on a real AMR code at 512--4096 CPU ranks against
    % TAU, Score-P, Caliper, and DFTracer.
    \textbf{Workload:} AMR code at 512--4096 CPU ranks (1 core/rank)
    \begin{itemize}
      \item AMR: Adaptive Mesh Refinement, highly dynamic simulations
      \item \textbf{0.8--2\%} runtime overhead vs.\ 56--81\% for existing tracing tools
      \item \textbf{3.6--44$\times$} smaller traces via Parquet columnar layout
      \item \textbf{3--4 orders of magnitude} faster post-mortem queries (zero-ETL)
    \end{itemize}

  \block{Closed-Loop Evaluation: LLM-driven Agentic Debugging}{
    \begin{itemize}
